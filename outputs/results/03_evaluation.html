def predict_single_image(image_path):
    
    # Load and preprocess
    img = load_img(image_path, target_size=IMAGE_SIZE)
    img_array = img_to_array(img) / 255.0
    img_array = np.expand_dims(img_array, axis=0)
    
    # Predict
    predictions = model.predict(img_array, verbose=0)
    predicted_idx = np.argmax(predictions[0])
    confidence = predictions[0][predicted_idx]
    predicted_name = class_names_short[predicted_idx]
    
    # Plot
    fig, axes = plt.subplots(1, 2, figsize=(14, 5))
    
    # Show image
    axes[0].imshow(img)
    axes[0].axis('off')
    axes[0].set_title(
        f'Prediction: {predicted_name}\nConfidence: {confidence:.2%}',
        fontsize=13, fontweight='bold',
        color='green' if confidence > 0.80 else 'orange'
    )
    
    # ✅ Fix — NUM_CLASSES defined separately before list comprehension
    NUM_CLASSES = len(class_names_short)
    colors = ['#2ecc71' if i == predicted_idx else '#3498db'
              for i in range(NUM_CLASSES)]
    
    axes[1].barh(class_names_short, predictions[0], color=colors)
    axes[1].set_xlabel('Confidence Score')
    axes[1].set_title('Confidence per Disease Class', fontsize=12)
    axes[1].axvline(x=0.5, color='red', linestyle='--', alpha=0.5)
    
    plt.tight_layout()
    plt.savefig("../outputs/single_prediction.png", dpi=150, bbox_inches='tight')
    plt.show()
    
    print(f"\n🍅 Prediction Result:")
    print(f"   Disease    : {predicted_name}")
    print(f"   Confidence : {confidence:.2%}")
    
    if confidence > 0.80:
        print("   Status     : ✅ High confidence prediction")
    elif confidence > 0.50:
        print("   Status     : 🟡 Moderate confidence — double check")
    else:
        print("   Status     : 🔴 Low confidence — model is unsure")
